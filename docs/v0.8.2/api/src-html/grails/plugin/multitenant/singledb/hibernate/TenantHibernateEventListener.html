024    /**
025     * Listens for pre-insert, pre-update and load / fetch Hibernate events. If the
026     * domain class related to the event is a multi-tenant class we apply the
027     * relevant constraints.
028     * 
029     * @author Kim A. Betti
030     */
031    @SuppressWarnings("serial")
032    public class TenantHibernateEventListener implements PreInsertEventListener, PreUpdateEventListener, LoadEventListener, PostLoadEventListener, PreDeleteEventListener {
033    
034        private static Logger log = LoggerFactory.getLogger(TenantHibernateFilterConfigurator.class);
035    
036        private CurrentTenant currentTenant;
037        private HibernateEventPropertyUpdater hibernateEventPropertyUpdater;
038    
039        /**
040         * One important thing to know. It's not enough to update the entity
041         * instance with the new tenant-id. Hibernate will _not_ pick up on this and
042         * will therefore not save the updated tenant id to database.
043         * 
044         * We have to get hold of the JPA meta model, find the index of the tenantId
045         * field and update the entity state in the event.
046         */
047        @Override
048        public boolean onPreInsert(PreInsertEvent event) {
049            if (isMultiTenantEntity(event.getEntity())) {
050                if (!currentTenant.isSet()) {
051                    throw new NoCurrentTenantException("Tried to save multi-tenant domain class '"
052                            + event.getEntity().getClass().getSimpleName() + "', but no tenant is set");
053                }
054    
055                Integer currentTenantId = currentTenant.get();
056                hibernateEventPropertyUpdater.updateProperty(event, MultiTenantAST.TENANT_ID_FIELD_NAME, currentTenantId);
057                MultiTenantDomainClass entity = (MultiTenantDomainClass) event.getEntity();
058                entity.setTenantId(currentTenantId);
059            }
060    
061            return false;
062        }
063    
064        /**
065         * TODO: Decide what to do in case tenant X attempts to update an entity
066         * without a tenant id. Should we automatically assign this entity to tenant X?
067         */
068        @Override
069        public boolean onPreUpdate(PreUpdateEvent event) {
070            if (isMultiTenantEntity(event.getEntity())) {
071                Integer currentTenantId = currentTenant.get();
072                MultiTenantDomainClass entity = (MultiTenantDomainClass) event.getEntity();
073                Integer entityTenantId = entity.getTenantId();
074                if (currentTenantId != null && !currentTenantId.equals(entityTenantId)) {
075                    throw new TenantSecurityException("Tried to update '" + event.getEntity() + "' with another tenant id. Expected "
076                            + currentTenantId + ", found " + entityTenantId, currentTenantId, entityTenantId);
077                }
078            }
079    
080            return false;
081        }
082    
083        @Override
084        public void onLoad(LoadEvent event, LoadType loadType) throws HibernateException {
085            Object LoadedEntity = event.getResult();
086            if (LoadedEntity != null && isMultiTenantEntity(LoadedEntity)) {
087                MultiTenantDomainClass entity = (MultiTenantDomainClass) LoadedEntity;
088                Integer currentTenantId = currentTenant.get();
089    
090                // We won't be able to extract tenant-id from an association fetch.
091                // TODO: This is a bit scary as it means that we potentially can load entities from
092                // other tenants through various variants of Hibernate collection / lazy loading.
093                if (!event.isAssociationFetch() && !allowEntityLoad(currentTenantId, entity)) {
094                    log.debug("Refusing tenant {} to load {}", currentTenant.get(), entity.getClass().getSimpleName());
095                    event.setResult(null);
096                }
097            }
098        }
099    
100        protected boolean allowEntityLoad(Integer currentTenantId, MultiTenantDomainClass entity) {
101            if (currentTenantId != null) {
102                if (belongsToCurrentTenant(currentTenantId, entity)) {
103                    return true;
104                } else {
105                    log.warn("Tried to fetch an instance of {} belonging to {}", entity.getClass().getName(), entity.getTenantId());
106                    return false;
107                }
108            } else {
109                return true; // No current tenant => no restrictions
110            }
111        }

140    
141        @Override
142        public boolean onPreDelete(PreDeleteEvent event) {
143            boolean shouldVetoDelete = false;
144            if (isMultiTenantEntity(event.getEntity())) {
145                MultiTenantDomainClass tenantEntity = (MultiTenantDomainClass) event.getEntity();
146                Integer currentTenantId = currentTenant.get();
147    
148                if (currentTenantId != null && !belongsToCurrentTenant(currentTenantId, tenantEntity)) {
149                    log.warn("Tenant {} tried to delete another tenants entity {}", currentTenant.get(), tenantEntity);
150                    shouldVetoDelete = true;
151                }
152            }
153    
154            return shouldVetoDelete;
155        }
156    
157        protected boolean belongsToCurrentTenant(Integer currentTenantId, MultiTenantDomainClass entity) {
158            Integer entityTenantId = entity.getTenantId();
159            return currentTenantId != null && currentTenantId.equals(entityTenantId);
160        }
161    
162        protected boolean isMultiTenantEntity(Object entity) {
163            return entity instanceof MultiTenantDomainClass;
164        }
165    
166        public void setCurrentTenant(CurrentTenant currentTenant) {
167            this.currentTenant = currentTenant;
168        }
169    
170        public void setHibernateEventPropertyUpdater(HibernateEventPropertyUpdater updater) {
171            this.hibernateEventPropertyUpdater = updater;
172        }
