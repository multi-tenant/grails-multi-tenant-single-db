001    package grails.plugin.multitenant.singledb.hibernate;
002    
003    import grails.plugin.hibernatehijacker.hibernate.HibernateConfigPostProcessor;
004    import grails.plugin.multitenant.core.MultiTenantDomainClass;
005    
006    import java.util.Iterator;
007    
008    import org.hibernate.HibernateException;
009    import org.hibernate.cfg.Configuration;
010    import org.hibernate.engine.FilterDefinition;
011    import org.hibernate.mapping.PersistentClass;
012    import org.slf4j.Logger;
013    import org.slf4j.LoggerFactory;
014    
015    /**
016     * Defines the Hibernate filter.
017     * 
018     * The Hibernate Hijacker detects beans implementing
019     * HibernateConfigPostProcessor and will make sure that the Configuration
020     * instances is passed through this instance when the application is starting.
021     * 
022     * @author Kim A. Betti <kim@developer-b.com>
023     */
024    public class TenantHibernateFilterConfigurator implements HibernateConfigPostProcessor {
025    
026        private static Logger log = LoggerFactory.getLogger(TenantHibernateFilterConfigurator.class);
027    
028        private FilterDefinition multiTenantHibernateFilter;
029    
030        @Override
031        public void doPostProcessing(Configuration configuration) throws HibernateException {
032            log.debug("Configuring multi-tenant Hibernate filter");
033            addMultiTenantFilterDefinition(configuration);
034            enrichMultiTenantDomainClasses(configuration);
035        }
036    
037        private void addMultiTenantFilterDefinition(Configuration configuration) {
038            log.debug("Defining Multi Tenant Hibernate filer");
039            configuration.addFilterDefinition(multiTenantHibernateFilter);
040        }
041    
042        @SuppressWarnings("unchecked")
043        private void enrichMultiTenantDomainClasses(Configuration configuration) {
044            Iterator<PersistentClass> mappingIterator = configuration.getClassMappings();
045            while (mappingIterator.hasNext()) {
046                PersistentClass persistentClass = mappingIterator.next();
047                if (isMultiTenantClass(persistentClass.getMappedClass())) {
048                    enrichMultiTenantDomainClass(persistentClass, configuration);
049                }
050            }
051        }
052    
053        private void enrichMultiTenantDomainClass(PersistentClass persistentClass, Configuration configuration) {
054            log.trace("Enabling multi-tenant mode for domain class {}", persistentClass.getClassName());
055            addDomainFilter(persistentClass, configuration);
056        }
057    
058        private boolean isMultiTenantClass(Class<?> mappedClass) {
059            return MultiTenantDomainClass.class.isAssignableFrom(mappedClass);
060        }
061    
062        private void addDomainFilter(PersistentClass persistentClass, Configuration configuration) {
063            String filterName = multiTenantHibernateFilter.getFilterName();
064            String condition = multiTenantHibernateFilter.getDefaultFilterCondition();
065            persistentClass.addFilter(filterName, condition);
066        }
067    
068        public void setMultiTenantHibernateFilter(FilterDefinition multiTenantHibernateFilter) {
069            this.multiTenantHibernateFilter = multiTenantHibernateFilter;
070        }
071    
072    }
