001    package grails.plugin.multitenant.singledb.hibernate;
002    
003    import grails.plugin.multitenant.core.Tenant;
004    import grails.plugins.hawkeventing.Event;
005    import grails.plugins.hawkeventing.EventBroker;
006    import grails.plugins.hawkeventing.EventConsumer;
007    import grails.util.GrailsNameUtils;
008    
009    import org.slf4j.Logger;
010    import org.slf4j.LoggerFactory;
011    import org.springframework.beans.factory.InitializingBean;
012    
013    /**
014     * Finds the Tenant class and adds Hibernate event listeners.
015     * Works as proxy translating Hibernate events to non-Hibernate
016     * specific events. This class will have to be moved if we want
017     * to make a -core plugin completely independent of Hibernate.
018     * 
019     * @author Kim A. Betti
020     */
021    public class TenantHibernateEventProxy implements InitializingBean {
022    
023        private final static Logger log = LoggerFactory.getLogger(TenantHibernateEventProxy.class);
024    
025        private EventBroker eventBroker;
026        private Class<? extends Tenant> tenantClass;
027    
028        @Override
029        public void afterPropertiesSet() {
030            if (tenantClass != null) {
031                log.info("Registering tenant event listeners for {}", tenantClass.getSimpleName());
032                registerTenantListeners(tenantClass);
033            } else {
034                log.warn("Tenant class not set. Add something like this to Config.groovy: multitenant.tenantClass = com.your-app.Customer");
035            }
036        }
037    
038        public void registerTenantListeners(Class<? extends Tenant> tenantDomainClass) {
039            log.debug("Subscribing to Hibernate events from {}", tenantDomainClass.getSimpleName());
040            String tenantClassPropertyName = GrailsNameUtils.getPropertyName(tenantDomainClass);
041            registerForPostInsertEvents(tenantClassPropertyName);
042            registerForPostDeleteEvents(tenantClassPropertyName);
043            registerForPostUpdateEvents(tenantClassPropertyName);
044        }
045    
046        private void registerForPostInsertEvents(String tenantClassPropertyName) {
047            EventConsumer proxy = new HawkEventProxy(eventBroker, "tenant.created");
048            eventBroker.subscribe("hibernate.postInsert." + tenantClassPropertyName, proxy);
049        }
050    
051        private void registerForPostDeleteEvents(String tenantClassPropertyName) {
052            EventConsumer proxy = new HawkEventProxy(eventBroker, "tenant.deleted");
053            eventBroker.subscribe("hibernate.postDelete." + tenantClassPropertyName, proxy);
054        }
055    
056        private void registerForPostUpdateEvents(String tenantClassPropertyName) {
057            EventConsumer proxy = new HawkEventProxy(eventBroker, "tenant.updated");
058            eventBroker.subscribe("hibernate.postUpdate." + tenantClassPropertyName, proxy);
059        }
060    
061        public void setEventBroker(EventBroker eventBroker) {
062            this.eventBroker = eventBroker;
063        }
064    
065        public void setTenantClass(Class<? extends Tenant> tenantClass) {
066            this.tenantClass = tenantClass;
067        }
068    
069    }
070    
071    /**
072     * Re-publishes events under a new name.
073     */
074    class HawkEventProxy implements EventConsumer {
075    
076        EventBroker eventBroker;
077        String eventName;
078    
079        public HawkEventProxy(EventBroker eventBroker, String eventName) {
080            this.eventBroker = eventBroker;
081            this.eventName = eventName;
082        }
083    
084        @Override
085        public void consume(Event event) {
086            eventBroker.publish(eventName, event.getPayload());
087        }
088    
089    }
