001    package grails.plugin.multitenant.core.exception;
002    
003    /**
004     * Thrown when a tenant can't be found.
005     * @author Kim A. Betti
006     */
007    @SuppressWarnings("serial")
008    public class TenantNotFoundException extends TenantException {
009    
010        private Integer missingTenantId;
011    
012        public TenantNotFoundException(Integer tenantId) {
013            super("Unable to find tenant with id " + tenantId);
014            this.missingTenantId = tenantId;
015        }
016    
017        public TenantNotFoundException(String message) {
018            super(message);
019        }
020    
021        public TenantNotFoundException(String message, Throwable cause) {
022            super(message, cause);
023        }
024    
025        public Integer getMissingTenantId() {
026            return missingTenantId;
027        }
028    
029    }
