001    package grails.plugin.multitenant.core;
002    
003    /**
004     * Most applications will have one domain class implementing
005     * this interface and the tenantId method will most likely
006     * return the database generated id to allow for fast lookups
007     * from second level cache.
008     * 
009     * @author Kim A. Betti
010     */
011    public interface Tenant {
012    
013        /**
014         * This interface will often be implemented Grails domain classes.
015         * Naming this method by Java bean convention (getTenantId) will cause
016         * GORM to mistake it for a property and fail.
017         * @return tenant id, will often be a database PK
018         */
019        Integer tenantId();
020    
021    }
