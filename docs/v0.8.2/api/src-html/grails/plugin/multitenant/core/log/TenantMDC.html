001    package grails.plugin.multitenant.core.log;
002    
003    import org.slf4j.MDC;
004    
005    import grails.plugin.multitenant.core.CurrentTenant;
006    import grails.plugins.hawkeventing.Event;
007    import grails.plugins.hawkeventing.annotation.Consuming;
008    import grails.plugins.hawkeventing.annotation.HawkEventConsumer;
009    
010    /**
011     * MDC / Mapped diagnostic context.
012     * 
013     * @see http://logback.qos.ch/manual/mdc.html
014     * @see http://www.slf4j.org/api/org/slf4j/MDC.html
015     * @see http://blog.oio.de/2010/11/23/usage-tracking-with-grails/
016     * @author Kim A. Betti
017     */
018    @HawkEventConsumer
019    public class TenantMDC {
020    
021        @Consuming(CurrentTenant.TENANT_AFTER_CHANGE_EVENT)
022        public void currentTenantUpdated(Event event) {
023            Integer tenantId = (Integer) event.getPayload();
024            String tenant = (tenantId == null) ? "no-tenant" : "tenant-" + tenantId;
025            MDC.put("tenant", tenant);
026        }
027    
028    }
