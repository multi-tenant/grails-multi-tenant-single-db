001    package grails.plugin.multitenant.core;
002    
003    /**
004     * All multi tenant domain classes should implement this interface. 
005     * If a @MultiTenant annotated domain class doesn't implement this
006     * interface it will be added by an AST transformation. 
007     * 
008     * There is no need to add the methods defined by this interface 
009     * manually as a tenantId property is added to all @MultiTenant annotated
010     * domain classes. Groovy will create the get and set method for us. 
011     * 
012     * @author Kim A. Betti
013     */
014    public interface MultiTenantDomainClass {
015    
016        Integer getTenantId();
017        
018        void setTenantId(Integer tenantId);
019        
020    }
