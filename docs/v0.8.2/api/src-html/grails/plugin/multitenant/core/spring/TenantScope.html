001    package grails.plugin.multitenant.core.spring;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import java.util.Map;
005    import java.util.concurrent.ConcurrentHashMap;
006    
007    import org.springframework.beans.factory.ObjectFactory;
008    import org.springframework.beans.factory.config.Scope;
009    import org.springframework.context.ApplicationContext;
010    import org.springframework.context.ApplicationContextAware;
011    
012    /**
013     * Custom Spring scope for per-tenant Spring beans.
014     * 
015     * Important: There is currently no listener making sure
016     * to remove any per-tenant beans when you delete a tenant.
017     * So if you delete a tenant this class will potentially
018     * hold a strong reference to tenant scoped beans belonging
019     * to the deleted tenant.
020     * 
021     * Important: This class is expected to be thread-safe!
022     * 
023     * @author Kim A. Betti
024     */
025    public class TenantScope implements Scope, ApplicationContextAware {
026    
027        public static final String NAME = "tenant";
028    
029        private CurrentTenant currentTenant;
030        private ApplicationContext applicationContext;
031    
032        private ConcurrentHashMap<Integer, Map<String, Object>> tenantBeanCache = new ConcurrentHashMap<Integer, Map<String, Object>>(50);
033    
034        /**
035         * Return the object with the given name from the underlying scope, creating
036         * it if not found in the underlying storage mechanism.
037         */
038        @Override
039        public Object get(String name, ObjectFactory<?> objectFactory) {
040            Integer tenantId = currentTenant.get();
041            if (tenantId == null) { // ConcurrentHashMap does not allow nulls, so mapping to -1
042                tenantId = new Integer(-1);
043            }
044            Map<String, Object> beanCache = getBeanCacheForTenant(tenantId);
045            if (!beanCache.containsKey(name)) {
046                beanCache.put(name, applicationContext.getBean(name + "_prototype"));
047            }
048    
049            return beanCache.get(name);
050        }
051    
052        private Map<String, Object> getBeanCacheForTenant(Integer tenantId) {
053            if (!tenantBeanCache.containsKey(tenantId)) {
054                tenantBeanCache.put(tenantId, new ConcurrentHashMap<String, Object>(10));
055            }
056    
057            return tenantBeanCache.get(tenantId);
058        }
059    
060        @Override
061        public String getConversationId() {
062            return "tenant-scope-" + currentTenant.get();
063        }
064    
065        /**
066         * Register a callback to be executed on destruction of the specified object
067         * in the scope (or at destruction of the entire scope, if the scope does
068         * not destroy individual objects but rather only terminates in its
069         * entirety).
070         */
071        @Override
072        public void registerDestructionCallback(String name, Runnable callback) {
073            throw new RuntimeException("registerDestructionCallback " + name + " is not implemented");
074        }
075    
076        /**
077         * Remove the object with the given name from the underlying scope.
078         */
079        @Override
080        public Object remove(String name) {
081            throw new RuntimeException("remove " + name + " is not implemented");
082        }
083    
084        /**
085         * Resolve the contextual object for the given key, if any.
086         */
087        @Override
088        public Object resolveContextualObject(String key) {
089            throw new RuntimeException("resolveContextualObject " + key + " is not implemented");
090        }
091    
092        public void setCurrentTenant(CurrentTenant currentTenant) {
093            this.currentTenant = currentTenant;
094        }
095    
096        @Override
097        public void setApplicationContext(ApplicationContext applicationContext) {
098            this.applicationContext = applicationContext;
099        }
100    
101    }
