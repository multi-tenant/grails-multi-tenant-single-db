001    package grails.plugin.multitenant.core;
002    
003    import grails.plugin.multitenant.core.exception.TenantNotFoundException;
004    
005    /**
006     * Most applications will probably provide an implementation of this interface.
007     * Having this contract makes it possible to look up tenant instances across
008     * plugins and applications in a type safe way.
009     * 
010     * @author Kim A. Betti
011     */
012    public interface TenantRepository {
013    
014        /**
015         * Look up a tenant instance associated with a given id.
016         * @param tenantId This will often be a database PK
017         * @return The tenant instance associated with the tenant id
018         * @throws TenantNotFoundException
019         */
020        Tenant findByTenantId(Integer tenantId) throws TenantNotFoundException;
021    
022    }
