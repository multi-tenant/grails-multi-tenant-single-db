001    package grails.plugin.multitenant.core.servlet;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import grails.plugin.multitenant.core.resolve.TenantResolver;
005    
006    import java.io.IOException;
007    
008    import javax.servlet.Filter;
009    import javax.servlet.FilterChain;
010    import javax.servlet.FilterConfig;
011    import javax.servlet.ServletContext;
012    import javax.servlet.ServletException;
013    import javax.servlet.ServletRequest;
014    import javax.servlet.ServletResponse;
015    import javax.servlet.http.HttpServletRequest;
016    
017    import org.springframework.context.ApplicationContext;
018    import org.springframework.web.context.support.WebApplicationContextUtils;
019    
020    /**
021     * This filter will be applied to incoming http requests.
022     * The current tenant id is looked up using the application
023     * provided tenantResolver and saved to a thread local storage.
024     * 
025     * The thread local storage is nulled out when other filters
026     * and servlet has processed the request.
027     * @author Kim A. Betti
028     */
029    public class CurrentTenantServletFilter implements Filter {
030    
031        protected CurrentTenant currentTenant; // thread local storage
032        protected TenantResolver tenantResolver; // provided by implementation
033    
034        @Override
035        public void init(FilterConfig fc) throws ServletException {
036            ServletContext servletContext = fc.getServletContext();
037            ApplicationContext ctx = WebApplicationContextUtils.getWebApplicationContext(servletContext);
038            currentTenant = (CurrentTenant) ctx.getBean("currentTenant");
039            tenantResolver = (TenantResolver) ctx.getBean("tenantResolver");
040        }
041    
042        @Override
043        public void doFilter(ServletRequest request, ServletResponse response,
044                FilterChain filterChain) throws IOException, ServletException {
045    
046            try {
047                if (request instanceof HttpServletRequest) {
048                    HttpServletRequest httpRequest = (HttpServletRequest) request;
049                    Integer currentTenantId = tenantResolver.resolve(httpRequest);
050                    currentTenant.set(currentTenantId);
051                }
052    
053                filterChain.doFilter(request, response);
054            } finally {
055                currentTenant.set(null);
056            }
057        }
058    
059        @Override
060        public void destroy() {
061        }
062    
063    }
