001    package grails.plugin.multitenant.core.spring;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import grails.plugin.multitenant.core.CurrentTenantAware;
005    
006    import org.springframework.beans.BeansException;
007    import org.springframework.beans.factory.config.BeanPostProcessor;
008    
009    /**
010     * Inject the currentTenant bean into beans
011     * implementing the CurrentTenantAware interface.
012     * @author Kim A. Betti
013     */
014    public class CurrentTenantAwarePostProcessor implements BeanPostProcessor {
015    
016        private CurrentTenant currentTenant;
017    
018        @Override
019        public Object postProcessBeforeInitialization(Object bean, String beanName) throws BeansException {
020            if (bean instanceof CurrentTenantAware) {
021                CurrentTenantAware currentTenantAwareBean = (CurrentTenantAware) bean;
022                currentTenantAwareBean.setCurrentTenant(currentTenant);
023            }
024    
025            return bean;
026        }
027    
028        @Override
029        public Object postProcessAfterInitialization(Object bean, String beanName) throws BeansException {
030            return bean;
031        }
032    
033        public void setCurrentTenant(CurrentTenant currentTenant) {
034            this.currentTenant = currentTenant;
035        }
036    
037    }
