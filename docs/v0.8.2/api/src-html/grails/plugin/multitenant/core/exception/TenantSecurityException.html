001    package grails.plugin.multitenant.core.exception;
002    
003    @SuppressWarnings("serial")
004    public class TenantSecurityException extends TenantException {
005    
006        private Integer currentTenantId, loadedTenantId;
007    
008        public TenantSecurityException(String message, Integer currentTenantId, Integer loadedTenantId) {
009            super(message);
010            this.currentTenantId = currentTenantId;
011            this.loadedTenantId = loadedTenantId;
012        }
013    
014        public Integer getCurrentTenantId() {
015            return currentTenantId;
016        }
017    
018        public Integer getLoadedTenantId() {
019            return loadedTenantId;
020        }
021    
022    }
