001    package grails.plugin.multitenant.core.spring;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    import org.apache.commons.logging.Log;
007    import org.apache.commons.logging.LogFactory;
008    import org.codehaus.groovy.grails.commons.spring.BeanConfiguration;
009    import org.codehaus.groovy.grails.commons.spring.DefaultBeanConfiguration;
010    import org.springframework.aop.scope.ScopedProxyFactoryBean;
011    import org.springframework.beans.BeansException;
012    import org.springframework.beans.factory.config.BeanDefinition;
013    import org.springframework.beans.factory.config.BeanFactoryPostProcessor;
014    import org.springframework.beans.factory.config.ConfigurableListableBeanFactory;
015    import org.springframework.beans.factory.support.BeanDefinitionRegistry;
016    import org.springframework.beans.factory.support.DefaultListableBeanFactory;
017    import org.springframework.beans.factory.support.GenericBeanDefinition;
018    
019    /**
020     * Creates a tenant scoped proxy around beans configured to be unique per tenant.
021     * In short terms, this works by lazily creating a new prototyped definition of 
022     * each per-tenant bean for each tenant. 
023     * @author Kim A. Betti
024     */
025    public class ConfiguredTenantScopedBeanProcessor implements BeanFactoryPostProcessor {
026    
027        private static final Log log = LogFactory.getLog(ConfiguredTenantScopedBeanProcessor.class);
028        private List<String> perTenantBeans;
029        
030        @Override
031        public void postProcessBeanFactory(ConfigurableListableBeanFactory beanFactory) throws BeansException {
032            BeanDefinitionRegistry appCtx = (DefaultListableBeanFactory) beanFactory;
033            for (String beanName : getPerTenantBeanNames(appCtx)) {
034                log.info("Setting tenant scope for: " + beanName);
035                BeanDefinition beanDef = appCtx.getBeanDefinition(beanName);
036                processBean(beanName, beanDef, appCtx);
037            }
038        }
039    
040        private void processBean(String beanName, BeanDefinition beanDef, BeanDefinitionRegistry appCtx) {
041            final String tenantScopedBeanName = "_" + beanName;
042            final String prototypeBeanName = tenantScopedBeanName + "_prototype";
043    
044            makeTenantScopedCopy(appCtx, beanDef, tenantScopedBeanName);
045            makePrototypeScopedCopy(appCtx, beanDef, prototypeBeanName);
046            replaceWithScopedProxy(appCtx, beanName, tenantScopedBeanName);
047        }
048    
049        private List<String> getPerTenantBeanNames(BeanDefinitionRegistry beanFactory) {
050            List<String> perTenantBeanNames = new ArrayList<String>();
051            for (String beanName : beanFactory.getBeanDefinitionNames()) {
052                BeanDefinition beanDefinition = beanFactory.getBeanDefinition(beanName);
053                
054                if (perTenantBeans.contains(beanName)) {
055                    perTenantBeanNames.add(beanName);
056                } else if (beanDefinition.getScope() == TenantScope.NAME) {
057                    perTenantBeanNames.add(beanName);
058                }
059            }
060    
061            return perTenantBeanNames;
062        }
063    
064        private BeanDefinition makeTenantScopedCopy(BeanDefinitionRegistry appCtx, BeanDefinition beanDef, String tenantScopedBeanName) {
065            BeanDefinition tenantScopedBeanDef = copyBeanDefiniton(appCtx, beanDef, tenantScopedBeanName);
066            tenantScopedBeanDef.setScope(TenantScope.NAME);
067            return tenantScopedBeanDef;
068        }
069    
070        private BeanDefinition makePrototypeScopedCopy(BeanDefinitionRegistry appCtx, BeanDefinition orgBeanDef, String prototypeBeanName) {
071            BeanDefinition prototypeDefinition = copyBeanDefiniton(appCtx, orgBeanDef, prototypeBeanName);
072            prototypeDefinition.setScope(BeanDefinition.SCOPE_PROTOTYPE);
073            return prototypeDefinition;
074        }
075    
076        private BeanDefinition copyBeanDefiniton(BeanDefinitionRegistry appCtx, BeanDefinition orgBeanDef, String newBeanName) {
077            BeanDefinition copiedBeanDefinition = new GenericBeanDefinition(orgBeanDef);
078            appCtx.registerBeanDefinition(newBeanName, copiedBeanDefinition);
079            return copiedBeanDefinition;
080        }
081    
082        private void replaceWithScopedProxy(BeanDefinitionRegistry appCtx, String beanName, String targetBeanName) {
083            BeanConfiguration scopedProxy = new DefaultBeanConfiguration(ScopedProxyFactoryBean.class);
084            scopedProxy.addProperty("targetBeanName", targetBeanName);
085    
086            BeanDefinition scopedBeanDefinition = scopedProxy.getBeanDefinition();
087            appCtx.registerBeanDefinition(beanName, scopedBeanDefinition);
088        }
089    
090        /**
091         * Usually configured in the Grails bean DSL
092         * @param perTenantBeans
093         */
094        public void setPerTenantBeans(List<String> perTenantBeans) {
095            this.perTenantBeans = perTenantBeans;
096        }
097    
098    }
