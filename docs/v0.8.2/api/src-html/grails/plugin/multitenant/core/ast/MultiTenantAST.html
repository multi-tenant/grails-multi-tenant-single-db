001    package grails.plugin.multitenant.core.ast;
002    
003    import grails.plugin.multitenant.core.MultiTenantDomainClass;
004    
005    import java.lang.reflect.Modifier;
006    
007    import org.codehaus.groovy.ast.ASTNode;
008    import org.codehaus.groovy.ast.ClassNode;
009    import org.codehaus.groovy.ast.expr.ConstantExpression;
010    import org.codehaus.groovy.ast.stmt.Statement;
011    import org.codehaus.groovy.control.CompilePhase;
012    import org.codehaus.groovy.control.SourceUnit;
013    import org.codehaus.groovy.grails.compiler.injection.GrailsASTUtils;
014    import org.codehaus.groovy.transform.ASTTransformation;
015    import org.codehaus.groovy.transform.GroovyASTTransformation;
016    
017    /**
018     * Adds a tenantId property to domain classes annotated MultiTenant.
019     */
020    @GroovyASTTransformation(phase = CompilePhase.CANONICALIZATION)
021    public class MultiTenantAST implements ASTTransformation {
022    
023        public final static String TENANT_ID_FIELD_NAME = "tenantId";
024    
025        // TODO: ConstantExpression.NULL would be better, but that leads to all sorts of crazy problems
026        // with default GORM constraints. It looks like it will become easier to hook into this in Grails 1.4:
027        // https://github.com/grails/grails-core/blob/master/grails-core/src/main/groovy/org/codehaus/groovy/grails/validation/ConstraintsEvaluatorFactoryBean.java
028        public final static Integer NO_TENANT_VALUE = Integer.MIN_VALUE;
029    
030        @Override
031        public void visit(ASTNode[] astNodes, SourceUnit sourceUnit) {
032            ClassNode annotatedClassNode = (ClassNode) astNodes[1];
033            addTenantPropertyIfNecessary(annotatedClassNode);
034            implementInterface(annotatedClassNode, MultiTenantDomainClass.class);
035        }
036    
037        private void addTenantPropertyIfNecessary(ClassNode classNode) {
038            if (!hasTenantIdProperty(classNode)) {
039                addTenantProperty(classNode);
040            }
041        }
042    
043        private boolean hasTenantIdProperty(ClassNode node) {
044            return GrailsASTUtils.hasOrInheritsProperty(node, "tenantId");
045        }
046    
047        private void addTenantProperty(ClassNode node) {
048            ClassNode integerType = new ClassNode(Integer.class);
049            ConstantExpression defaultValue = new ConstantExpression(NO_TENANT_VALUE);
050            Statement getterBlock = null;
051            Statement setterBlock = null;
052            node.addProperty(TENANT_ID_FIELD_NAME, Modifier.PUBLIC,
053                    integerType, defaultValue, getterBlock, setterBlock);
054        }
055    
056        private void implementInterface(ClassNode node, Class<?> interf) {
057            ClassNode interfaceClassNode = new ClassNode(interf);
058            if (!node.implementsInterface(interfaceClassNode)) {
059                node.addInterface(interfaceClassNode);
060            }
061        }
062    
063    }
