001    package grails.plugin.multitenant.singledb.hibernate;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import grails.plugins.hawkeventing.Event;
005    import grails.plugins.hawkeventing.annotation.Consuming;
006    import grails.plugins.hawkeventing.annotation.HawkEventConsumer;
007    
008    import java.util.Set;
009    
010    import org.hibernate.SessionFactory;
011    import org.hibernate.classic.Session;
012    import org.hibernate.engine.FilterDefinition;
013    import org.springframework.transaction.support.TransactionSynchronizationManager;
014    
015    /**
016     * Subscribes itself to hibernate.sessionCreated events. Enables the tenant
017     * Hibernate filter with the current tenant id (if any).
018     *
019     * Important: tenantId = null will disable the Hibernate filter!
020     * @author Kim A. Betti
021     */
022    @HawkEventConsumer
023    public class TenantHibernateFilterEnabler {
024    
025        private SessionFactory sessionFactory;
026        private CurrentTenant currentTenant;
027        private FilterDefinition multiTenantHibernateFilter;
028    
029        @Consuming("hibernate.sessionCreated")
030        public void newHibernateSessionCreated(Event event) {
031            Session newSession = (Session) event.getPayload();
032            Integer currentTenantId = currentTenant.get();
033            updateFilterParameter(newSession, currentTenantId);
034        }
035    
036        @Consuming(CurrentTenant.TENANT_AFTER_CHANGE_EVENT)
037        public void currentTenantUpdated(Event event) {
038            if (hasSessionBoundToThread()) {
039                Integer updatedTenantId = (Integer) event.getPayload();
040                Session currentSession = sessionFactory.getCurrentSession();
041                updateFilterParameter(currentSession, updatedTenantId);
042            }
043        }
044    
045        private boolean hasSessionBoundToThread() {
046            return TransactionSynchronizationManager.hasResource(sessionFactory);
047        }
048    
049        private void updateFilterParameter(Session session, Integer tenantId) {
050            if (tenantId != null) {
051                enableHibernateFilterForTenant(session, tenantId);
052            } else {
053                disableHibernateFilter(session);
054            }
055        }
056    
057        @SuppressWarnings("unchecked")
058        private void enableHibernateFilterForTenant(Session session, Integer tenantId) {
059            String filterName = multiTenantHibernateFilter.getFilterName();
060            Set<String> paramNames = multiTenantHibernateFilter.getParameterNames();
061            String paramName = paramNames.iterator().next();
062            session.enableFilter(filterName).setParameter(paramName, tenantId);
063        }
064    
065        private void disableHibernateFilter(Session session) {
066            String filterName = multiTenantHibernateFilter.getFilterName();
067            session.disableFilter(filterName);
068        }
069    
070        public void setCurrentTenant(CurrentTenant currentTenant) {
071            this.currentTenant = currentTenant;
072        }
073    
074        public void setSessionFactory(SessionFactory sessionFactory) {
075            this.sessionFactory = sessionFactory;
076        }
077    
078        public void setMultiTenantHibernateFilter(FilterDefinition multiTenantHibernateFilter) {
079            this.multiTenantHibernateFilter = multiTenantHibernateFilter;
080        }
081    
082    }
