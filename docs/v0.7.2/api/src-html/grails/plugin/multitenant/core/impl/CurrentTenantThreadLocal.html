001    package grails.plugin.multitenant.core.impl;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import grails.plugins.hawkeventing.EventBroker;
005    
006    /**
007     * Stores the current tenant id on the current thread.
008     * Have a look at the link below if you want more information on ThreadLocal.
009     * @see http://stackoverflow.com/questions/1490919/purpose-of-threadlocal
010     */
011    public class CurrentTenantThreadLocal implements CurrentTenant {
012    
013        private static ThreadLocal<Integer> currentTenant = new ThreadLocal<Integer>();
014        private EventBroker eventBroker;
015    
016        @Override
017        public Integer get() {
018            return currentTenant.get();
019        }
020    
021        @Override
022        public void set(Integer tenantId) {
023            eventBroker.publish(TENANT_BEFORE_CHANGE_EVENT, tenantId);
024            currentTenant.set(tenantId);
025            eventBroker.publish(TENANT_AFTER_CHANGE_EVENT, tenantId);
026        }
027    
028        public void setEventBroker(EventBroker broker) {
029            this.eventBroker = broker;
030        }
031    
032    }
