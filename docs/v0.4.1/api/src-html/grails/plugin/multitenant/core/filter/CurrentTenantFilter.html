001    package grails.plugin.multitenant.core.filter;
002    
003    import grails.plugin.multitenant.core.CurrentTenant;
004    import grails.plugin.multitenant.core.resolve.TenantResolver;
005    
006    import java.io.IOException;
007    
008    import javax.servlet.Filter;
009    import javax.servlet.FilterChain;
010    import javax.servlet.FilterConfig;
011    import javax.servlet.ServletContext;
012    import javax.servlet.ServletException;
013    import javax.servlet.ServletRequest;
014    import javax.servlet.ServletResponse;
015    import javax.servlet.http.HttpServletRequest;
016    
017    import org.springframework.context.ApplicationContext;
018    import org.springframework.web.context.support.WebApplicationContextUtils;
019    
020    /**
021     * This filter will be applied to incoming http requests.
022     * The current tenant id is looked up using the application
023     * provided tenantResolver and saved to a thread local storage.
024     * 
025     * The thread local storage is nulled out when other filters
026     * and servlet has processed the request.
027     * 
028     * TODO: Rename package to servletfilters (or similar) to avoid confusion with Hibernate filters
029     * @author Kim A. Betti
030     */
031    public class CurrentTenantFilter implements Filter {
032    
033        private CurrentTenant currentTenant; // thread local storage
034        private TenantResolver tenantResolver; // provided by implementation
035    
036        @Override
037        public void init(FilterConfig fc) throws ServletException {
038            ServletContext servletContext = fc.getServletContext();
039            ApplicationContext ctx = WebApplicationContextUtils.getWebApplicationContext(servletContext);
040            currentTenant = (CurrentTenant) ctx.getBean("currentTenant");
041            tenantResolver = (TenantResolver) ctx.getBean("tenantResolver");
042        }
043    
044        @Override
045        public void doFilter(ServletRequest request, ServletResponse response,
046                FilterChain filterChain) throws IOException, ServletException {
047    
048            try {
049                if (request instanceof HttpServletRequest) {
050                    HttpServletRequest httpRequest = (HttpServletRequest) request;
051                    int currentTenantId = tenantResolver.resolve(httpRequest);
052                    currentTenant.set(currentTenantId);
053                }
054    
055                filterChain.doFilter(request, response);
056            } finally {
057                currentTenant.set(null);
058            }
059        }
060    
061        @Override
062        public void destroy() {
063        }
064    
065    }
